## T1
直接模拟

**代码：**
```cpp
#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,m;
int cp=0,tmp;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",&tmp);
		cp+=tmp;
		printf("%d ",cp/m);
		cp%=m;
	}

	return 0;
}
```
