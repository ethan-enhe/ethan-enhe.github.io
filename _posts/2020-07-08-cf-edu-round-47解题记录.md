## T1
直接按照题目要求模拟，复杂度$$O(M+N)$$
**代码**
```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int NR=1005;
int n,m;
int a[NR],b[NR];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)scanf("%d",a+i);
	for(int i=1;i<=m;i++)scanf("%d",b+i);
	int j=1;
	for(int i=1;(j<=n)&&(i<=n);i++)
		if(a[i]<=b[j])j++;
	printf("%d",j-1);
	return 0;
}

```
